# 코드

```c++

pair<vec, vec> rotating_calipers(vector<vec>& p) {
    vector<int> tmp = convex_hull(p);
    vector<vec> cx;
    for(int now : tmp) cx.push_back(p[now]);
    int p1 = 0, p2 = 1;
    long double maxv = (cx[p1] - cx[p2]).norm();
    ans1 = cx[p1];
    ans2 = cx[p2];
    
    for(int i=0; i<cx.size()*2; i++) {
        int np1 = (p1+1) % cx.size();
        int np2 = (p2+1) % cx.size();
        ll tmp = ccw(cx[np1] - cx[p1], cx[p2] - cx[np2]);
        if(tmp > 0) p1 = np1;
        if(tmp < 0) p2 = np2;
        if(tmp == 0) {
            p1 = np1;
            p2 = np2;
        }
        
        long double nowv = (cx[p1] - cx[p2]).norm();
        if(maxv < nowv) {
            maxv = nowv;
            ans1 = cx[p1];
            ans2 = cx[p2];
        }
    }
    
    return {ans1, ans2};
}
```
