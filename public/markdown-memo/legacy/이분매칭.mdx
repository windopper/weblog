using namespace std;
int aMatch[51];
int bMatch[51];
int dis[51];

bool dfs(int cur) {
	if(dis[cur]) return false;
	dis[cur] = true;
	for(int next : adj[cur]) {
		if(bMatch[next] == -1 || dfs(bMatch[next])) {
			bMatch[next] = cur;
			aMatch[cur] = next;
			return true;
		}
	}
	return false;
}
